	$("#floodgate").css('display', 'none');

	var currentFloodgateItems = null;
	var currentFloodgateIndex = 0;
	var lastFloodgateUpdate = Math.floor(Date.now() / 1000);
	
	function updateFloodgates() {
		if(currentFloodgateItems == null || (Math.floor(Date.now() / 1000) - lastFloodgateUpdate) > 120 ) { 
			lastFloodgateUpdate = Math.floor(Date.now() / 1000);
			currentFloodgateIndex = 0;
			$.ajax("http://www.511virginia.org/rssfeed/emergency.xml", {
				contentType:"application/xml",
				success:function(xml) {
					currentFloodgateItems = $(xml).find( "item" );
					if(currentFloodgateItems.length==0) {
						$("#floodgate").css('display', 'none');
					} else {
						$("#floodgate").css('display', '');
						var itemTitle = $(currentFloodgateItems[currentFloodgateIndex]).find('title');
						var itemLink = $(currentFloodgateItems[currentFloodgateIndex]).find('link');

						//if IE paradigm, else sane browsers
						if(itemTitle[0] && itemTitle[0].textContent) {
							$('#floodgate').empty().append('<div style="font-size:1.16em;"><a href="' + itemLink[0].textContent  + '">' + itemTitle[0].textContent + ' </a></div>')
						} else {
							$('#floodgate').empty().append('<div style="font-size:1.16em;"><a href="' + $(itemLink)[0].innerHTML  + '">' + $(itemTitle)[0].innerHTML + ' </a></div>')
						}
						currentFloodgateIndex++;
					}
				},
				error:function() {
						$("#floodgate").css('display', 'none');
				}
			})
		} else {
			if(currentFloodgateItems == null || currentFloodgateItems.length == 0) {
				return;
			}

			if(currentFloodgateIndex == currentFloodgateItems.length) {
				currentFloodgateIndex = 0;
			}
			
			var itemTitle = $(currentFloodgateItems[currentFloodgateIndex]).find('title');
			var itemLink = $(currentFloodgateItems[currentFloodgateIndex]).find('link');

			//if IE paradigm, else sane browsers
			if(itemTitle[0] && itemTitle[0].textContent) {
				$('#floodgate').empty().append('<div style="font-size:1.16em;"><a href="' + itemLink[0].textContent  + '">' + itemTitle[0].textContent + ' </a></div>')
			} else {
				$('#floodgate').empty().append('<div style="font-size:1.16em;"><a href="' + $(itemLink)[0].innerHTML  + '">' + $(itemTitle)[0].innerHTML + ' </a></div>')
			}

			currentFloodgateIndex++;
		}
	};

	setInterval(updateFloodgates, 10000);

        replaceCurrent ? location.replace(url) : window.open(url);
